python中的urllib2

MarkdownHTML
> 本篇介绍python中的urllib2模块


python3中urllib2已经被拆成了urllib.request urllib.error两个模块
urllib2提供了基本的认证，重定向，cookie等url相关的函数和类

## 使用

### 1.常用方法
* `urllib2.urlopen(url,data,timeout) raise URLError`
* `urllib2.install_opener(opener)`设置一个OpenerDirector作为全局默认opener
* `urllib2.build_opener([handler,...])` 返回一个OpenerDirector实例；自带有ProxyHandler,unknownHandler,HttpHandler,HttpDefaultErrorHandler,HttpRedirectHandler,FtpHandler,HttpErrorProcessor,HttpsHandler
* `urllib2.Request(url,data,headers,origin_req_host,unverifiable)`


### 2.类
* BaseHandler
* HTTPRedirectHandler
* HTTPCookieProcessor
* ProxyHandler
* HTTPPasswordMgr
* HTTPBasicAuthHandler
* ProxyBasicAuthHandler
* HTTPHandler
* HTTPSHandler
* FileHandler
* FTPHandler
* CacheFTPHandler
* UnknownHandler
* HTTPErrorProcessor

## 代码example

```
# http://cuiqingcai.com/1052.html
# https://docs.python.org/2/library/urllib2.html#module-urllib2

#使用urllib2 urllib包进行请求
import urllib
import urllib2
url='http://www.baidu.com'
user_agent = 'Mozilla/4.0 (compatible; MSIE 5.5; Windows NT)'
values = {'name':'gong','age':22}
headers = {'User-Agent':user_agent,'Referer':'http://www.zhihu.com/articles'}
data = urllib.urlencode(values)

request = urllib2.Request(url,data,headers)
#-------------------------------------------------------------------------------
#动态增加header
import urllib2
req = urllib2.Request('http://www.example.com/')
req.add_header('Referer', 'http://www.python.org/')
# Customize the default User-Agent header value:
req.add_header('User-Agent', 'urllib-example/0.1 (Contact: . . .)')
r = urllib2.urlopen(req)
#------------------------------------------------------------------------------
#代理的使用
import urllib2
enable_proxy = True
proxy_handler = urllib2.ProxyHandler({'http':'http://192.168.5.190:8080'})
null_proxy_handler = urllib2.ProxyHandler({})
if enable_proxy:
    opener = urllib2.build_opener(proxy_handler)
else:
    opener = urllib2.bulid_opener(null_proxy_handler)
urllib2.install_opener(opener)

#-------------------------------------------------------------------------------

proxy_handler = urllib2.ProxyHandler({'http': 'http://www.example.com:3128/'})
proxy_auth_handler = urllib2.ProxyBasicAuthHandler()
proxy_auth_handler.add_password('realm', 'host', 'username', 'password')

opener = urllib2.build_opener(proxy_handler, proxy_auth_handler)
# This time, rather than install the OpenerDirector, we use it directly:
opener.open('http://www.example.com/login.html')
#-------------------------------------------------------------------------------
#URLError
import urllib2
request = urllib2.Request('http://xxx.com/do-not-exist')
try:
    urllib2.urlopen(request)
except urllib2.URLError,e:
    print e.reason

# urllib2.HTTPError ,e
# HTTPError是URLError的子类
# print e.code

#-------------------------------------------------------------------------------
#cookie的使用
import urllib2
import cookielib
cookie = cookielib.CookieJar()
handler = urllib2.HTTPCookieProcessor(cookie)
opener = urllib2.bulid_opener(handler)
response = opener.open('http://www.baide.com')
for item in cookie:
    print item.name,":",item.value
# cookie.save()
# cookie.load()

#------------------------------------------------------------------------------
#basic http 验证
import urllib2
# Create an OpenerDirector with support for Basic HTTP Authentication...
auth_handler = urllib2.HTTPBasicAuthHandler()
auth_handler.add_password(realm='PDQ Application',
                          uri='https://mahler:8092/site-updates.py',
                          user='klem',
                          passwd='kadidd!ehopper')
opener = urllib2.build_opener(auth_handler)
# ...and install it globally so it can be used with urlopen.
urllib2.install_opener(opener)
urllib2.urlopen('http://www.example.com/login.html')

#-------------------------------------------------------------------------------
#opener 层面为所有的request都增加headers
import urllib2
opener = urllib2.build_opener()
opener.addheaders = [('User-agent', 'Mozilla/5.0')]
opener.open('http://www.example.com/')
```
